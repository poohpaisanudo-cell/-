1. Pseudocode คำนวณพื้นที่วงกลม

BEGIN
    DISPLAY "Enter radius: "
    INPUT radius
    SET area = π × radius × radius
    DISPLAY "Area of circle = ", area
END

2. Pseudocode คำนวณเกรด

BEGIN
    DISPLAY "Enter score (0-100): "
    INPUT score
    IF score >= 80 THEN
        SET grade = "A"
    ELSE IF score >= 70 THEN
        SET grade = "B"
    ELSE IF score >= 60 THEN
        SET grade = "C"
    ELSE IF score >= 50 THEN
        SET grade = "D"
    ELSE
        SET grade = "F"
    END IF
    DISPLAY "Grade = ", grade
END

3. Pseudocode หาค่าสูงสุดใน 3 จำนวน

BEGIN
    DISPLAY "Enter 3 numbers: "
    INPUT num1, num2, num3
    SET max = num1
    IF num2 > max THEN
        SET max = num2
    END IF
    IF num3 > max THEN
        SET max = num3
    END IF
    DISPLAY "Maximum = ", max
END
